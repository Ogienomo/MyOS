   211→// ─── NUDGE SYSTEM: Calculate overdue check-ins with escalation ──
   212→interface NudgeInfo {
   213→  type: string
   214→  label: string
   215→  emoji: string
   216→  minutesOverdue: number
   217→  level: 'warning' | 'urgent' | 'critical'
   218→  message: string

   221→function getOverdueNudges(completedCheckIns: string[], currentTime: Date): NudgeInfo[] {
   222→  const hour = currentTime.getHours()
   223→  const minute = currentTime.getMinutes()
   224→  const now = hour * 60 + minute
   225→  const nudges: NudgeInfo[] = []
   226→
   227→  // Morning: window closes at 10 AM (600 min), overdue after 10 AM
   228→  if (!completedCheckIns.includes('morning') && now >= 600) {
   229→    const minutesOverdue = now - 600
   230→    if (minutesOverdue > 0) {
   231→      nudges.push({
   232→        type: 'morning',
   233→        label: 'Morning Alignment',
   234→        emoji: '',
   235→        minutesOverdue,
   236→        level: minutesOverdue >= 120 ? 'critical' : minutesOverdue >= 60 ? 'urgent' : 'warning',
   237→        message: minutesOverdue >= 120
   238→          ? `Your morning check-in is ${Math.floor(minutesOverdue / 60)}h ${minutesOverdue % 60}m overdue. This is critical, Praise.`
   239→          : minutesOverdue >= 60
   240→            ? `Morning check-in is ${Math.floor(minutesOverdue / 60)}h overdue. Stop everything and do it now.`
   241→            : `Morning check-in is ${minutesOverdue}m overdue. Don't let this slip.`,
   242→      })
   243→    }
   244→  }
   245→
   246→  // Midday: window closes at 14:00 (840 min), overdue after 14:00
   247→  if (!completedCheckIns.includes('midday') && now >= 840) {
   248→    const minutesOverdue = now - 840
   249→    if (minutesOverdue > 0) {
   250→      nudges.push({
   251→        type: 'midday',
   252→        label: 'Midday Correction',
   253→        emoji: '',
   254→        minutesOverdue,
   255→        level: minutesOverdue >= 120 ? 'critical' : minutesOverdue >= 60 ? 'urgent' : 'warning',
   256→        message: minutesOverdue >= 120
   257→          ? `Midday check-in is ${Math.floor(minutesOverdue / 60)}h ${minutesOverdue % 60}m overdue. The day is slipping.`
   258→          : minutesOverdue >= 60
   259→            ? `Midday check-in is ${Math.floor(minutesOverdue / 60)}h overdue. Course-correct now.`
   260→            : `Midday check-in is ${minutesOverdue}m overdue. Quick check-in to realign.`,
   261→      })
   262→    }
   263→  }
   264→
   265→  // Evening: window closes at 22:00 (1320 min), overdue after 22:00
   266→  if (!completedCheckIns.includes('evening') && now >= 1320) {
   267→    const minutesOverdue = now - 1320
   268→    if (minutesOverdue > 0) {
   269→      nudges.push({
   270→        type: 'evening',
   271→        label: 'Evening Review',
   272→        emoji: '',
   273→        minutesOverdue,
   274→        level: minutesOverdue >= 120 ? 'critical' : minutesOverdue >= 60 ? 'urgent' : 'warning',
   275→        message: minutesOverdue >= 120
   276→          ? `Evening review is ${Math.floor(minutesOverdue / 60)}h overdue. Skipping accountability is not an option.`
   277→          : minutesOverdue >= 60
   278→            ? `Evening review is ${Math.floor(minutesOverdue / 60)}h overdue. Do it before you sleep.`
   279→            : `Evening review is ${minutesOverdue}m overdue. Account for your day now.`,
   280→      })
   281→    }
   282→  }
   283→
   284→  // Sort by most critical first
   285→  return nudges.sort((a, b) => {
   286→    const levelOrder = { critical: 3, urgent: 2, warning: 1 }
   287→    return levelOrder[b.level] - levelOrder[a.level]
   288→  })
   289→}

   292→function formatTime(date: Date): string {
   293→  let hours = date.getHours()
   294→  const minutes = date.getMinutes()
   295→  const ampm = hours >= 12 ? 'PM' : 'AM'
   296→  hours = hours % 12
   297→  if (hours === 0) hours = 12
   298→  return `${hours}:${minutes.toString().padStart(2, '0')}\u00A0${ampm}`
   299→}

   301→function formatMinutesToTime(minutes: number): string {
   302→  const h = Math.floor(minutes / 60) % 24
   303→  const m = minutes % 60
   304→  const ampm = h >= 12 ? 'PM' : 'AM'
   305→  const displayH = h === 0 ? 12 : h > 12 ? h - 12 : h
   306→  return `${displayH}:${m.toString().padStart(2, '0')}\u00A0${ampm}`
   307→}

   309→function getTimezoneAbbr(): string {
   310→  try {
   311→    const formatter = new Intl.DateTimeFormat('en-US', {
   312→      timeZone: USER_TIMEZONE,
   313→      timeZoneName: 'short',
   314→    })
   315→    const parts = formatter.formatToParts(new Date())
   316→    const tzPart = parts.find(p => p.type === 'timeZoneName')
   317→    return tzPart?.value || 'WAT'
   318→  } catch {
   319→    return 'WAT'
   320→  }
   321→}

   323→function formatCountdown(ms: number): string {
   324→  if (ms <= 0) return 'Closing now'
   325→  const totalMinutes = Math.floor(ms / 60000)
   326→  const hours = Math.floor(totalMinutes / 60)
   327→  const minutes = totalMinutes % 60
   328→  if (hours > 0) return `${hours}h ${minutes}m left`
   329→  if (minutes > 0) return `${minutes}m left`
   330→  return 'Less than 1m left'
   331→}
