    67→        count: entries.length,
    68→      }
    69→    })(),

    72→

    87→function formatContextForAI(ctx: LifeContext): string {
    88→  const parts: string[] = []
    89→
    90→  if (ctx.recentCheckIns.length > 0) {
    91→    parts.push('--- RECENT CHECK-INS ---')
    92→    for (const ci of ctx.recentCheckIns) {
    93→      parts.push(`[${ci.type}] ${ci.date}: ${ci.data.substring(0, 300)}`)
    94→    }
    95→  }
    96→  if (ctx.recentScores.length > 0) {
    97→    parts.push('--- RECENT LIFE SCORES ---')
    98→    for (const s of ctx.recentScores) {
    99→      parts.push(`${s.date}: Faith=${s.faith} Health=${s.health} Career=${s.career} Business=${s.havilah} Finances=${s.finances} Relationships=${s.relationships} PersonalGrowth=${s.personalGrowth} Overall=${s.overall}`)
   100→    }
   101→  }
   102→  if (ctx.activeGoals.length > 0) {
   103→    parts.push('--- ACTIVE GOALS ---')
   104→    for (const g of ctx.activeGoals) {
   105→      const completedTasks = g.tasks.filter(t => t.status === 'Completed').length
   106→      parts.push(`[${g.area}] ${g.title} (${completedTasks}/${g.tasks.length} tasks done)`)
   107→    }
   108→  }
   109→  if (ctx.activeAlerts.length > 0) {
   110→    parts.push('--- ACTIVE DRIFT ALERTS ---')
   111→    for (const a of ctx.activeAlerts) {
   112→      parts.push(`[${a.severity}] ${a.area}: ${a.message}`)
   113→    }
   114→  }
   115→  if (ctx.recentMemories.length > 0) {
   116→    parts.push('--- RECENT MEMORIES ---')
   117→    for (const m of ctx.recentMemories) {
   118→      parts.push(`[${m.type}] ${m.area}: ${m.content}`)
   119→    }
   120→  }
   121→  if (ctx.recentMoodLogs.length > 0) {
   122→    parts.push('--- RECENT MOOD LOGS ---')
   123→    for (const l of ctx.recentMoodLogs) {
   124→      parts.push(`${l.date} ${l.time}: mood=${l.mood} energy=${l.energy} focus=${l.focus}${l.note ? ` "${l.note}"` : ''}`)
   125→    }
   126→  }
   127→  if (ctx.recentFinances.length > 0) {
   128→    parts.push('--- RECENT FINANCE ENTRIES ---')
   129→    for (const f of ctx.recentFinances) {
   130→      parts.push(`${f.date}: ${f.type} ₦${f.amount} (${f.category})${f.purpose ? ` — ${f.purpose}` : ''}`)
   131→    }
   132→  }
   133→  if (ctx.weekFinances.count > 0) {
   134→    parts.push('--- THIS WEEK FINANCES ---')
   135→    parts.push(`Received: ₦${ctx.weekFinances.received.toLocaleString()} | Spent: ₦${ctx.weekFinances.spent.toLocaleString()} | Net: ₦${ctx.weekFinances.net.toLocaleString()}`)
   136→  }
   137→  if (ctx.lifeAreaProgress.length > 0) {
   138→    parts.push('--- LIFE AREA PROGRESS ---')
   139→    for (const lap of ctx.lifeAreaProgress) {
   140→      parts.push(`[${lap.area}] Current: ${lap.currentStatus || 'Not set'} | Vision: ${lap.idealVision || 'Not set'}${lap.blockers ? ` | Blockers: ${lap.blockers}` : ''}`)
   141→    }
   142→  }
   143→  parts.push(`--- CURRENT DATE: ${ctx.today} ---`)
   144→  // Include the current time + time-of-day label so the AI is always
   145→  // time-aware. Without this, the AI gives generic advice that doesn't match
   146→  // the time (e.g. "execute today's plan" at 11:50 PM).
   147→  // IMPORTANT: Use timezone-aware functions — `new Date().getHours()` returns
   148→  // server-local time (UTC in production), which is off by 1 hour from the
   149→  // user's timezone (Africa/Lagos, UTC+1).
   150→  const hours = getCurrentHourInTimezone()
   151→  const timeStr = getCurrentTimeStringInTimezone()
   152→  let timeLabel: string
   153→  if (hours < 10) timeLabel = 'MORNING (Morning Alignment window — start the day)'
   154→  else if (hours < 14) timeLabel = 'MIDDAY (Midday Correction window — course-correct)'
   155→  else if (hours < 17) timeLabel = 'AFTERNOON (execution time — push through)'
   156→  else if (hours < 21) timeLabel = 'EVENING (Evening Review window — account for the day)'
   157→  else if (hours < 23) timeLabel = 'LATE EVENING (reflect + prepare for tomorrow — do NOT say "execute today\'s plan")'
   158→  else timeLabel = 'NIGHT (Praise should be resting — focus on tomorrow, NOT today. Do NOT say "execute today\'s plan")'
   159→  parts.push(`--- CURRENT TIME: ${timeStr} (Africa/Lagos) ---`)
   160→  parts.push(`--- TIME-OF-DAY: ${timeLabel} ---`)
   161→
   162→  return parts.join('\n')
   163→}

   165→// Server-side strict mode check
   166→function isWithinCheckInWindowServer(checkInType: string, settings: { strictMode: boolean; morningEnabled: boolean; morningTime: string; eveningEnabled: boolean; eveningTime: string; windowMinutes: number }): { allowed: boolean; message: string | null } {
   167→  if (!settings.strictMode) return { allowed: true, message: null }
   168→  if (!['morning', 'evening'].includes(checkInType)) return { allowed: true, message: null }
   169→
   170→  // Use timezone-aware current minutes — server-local time (UTC in prod)
   171→  // would misalign check-in windows by the timezone offset.
   172→  const currentMinutes = getCurrentMinutesInTimezone()
   173→
   174→  if (checkInType === 'morning') {
   175→    if (!settings.morningEnabled) return { allowed: false, message: 'Morning check-in is disabled in your settings.' }
   176→    const [h, m] = settings.morningTime.split(':').map(Number)
   177→    const targetMinutes = h * 60 + m
   178→    const windowMin = Math.max(0, targetMinutes - settings.windowMinutes)
   179→    const windowMax = targetMinutes + settings.windowMinutes
   180→    if (currentMinutes >= windowMin && currentMinutes <= windowMax) return { allowed: true, message: null }
   181→    const nextH = Math.floor(windowMin / 60) % 24
   182→    const nextM = windowMin % 60
   183→    return { allowed: false, message: `Check-in window is closed. Your next morning check-in opens at ${nextH.toString().padStart(2, '0')}:${nextM.toString().padStart(2, '0')}. Strict mode is ON.` }
   184→  }
   185→
   186→  if (checkInType === 'evening') {
   187→    if (!settings.eveningEnabled) return { allowed: false, message: 'Evening check-in is disabled in your settings.' }
   188→    const [h, m] = settings.eveningTime.split(':').map(Number)
   189→    const targetMinutes = h * 60 + m
   190→    const windowMin = Math.max(0, targetMinutes - settings.windowMinutes)
   191→    const windowMax = targetMinutes + settings.windowMinutes
   192→    if (currentMinutes >= windowMin && currentMinutes <= windowMax) return { allowed: true, message: null }
   193→    const nextH = Math.floor(windowMin / 60) % 24
   194→    const nextM = windowMin % 60
   195→    return { allowed: false, message: `Check-in window is closed. Your next evening check-in opens at ${nextH.toString().padStart(2, '0')}:${nextM.toString().padStart(2, '0')}. Strict mode is ON.` }
   196→  }
   197→
   198→  return { allowed: true, message: null }
   199→}

   207→  faith: 'Faith & Spiritual Life',

   216→function analyzeScores(ctx: LifeContext): { lowest: string[]; highest: string[]; trend: string; analysis: string } {
   217→  if (ctx.recentScores.length === 0) {
   218→    return { lowest: [], highest: [], trend: 'no_data', analysis: 'No scores recorded yet. Start by scoring your life areas to get personalized insights.' }
   219→  }
   220→
   221→  const latest = ctx.recentScores[0]
   222→  const areas = ['faith', 'health', 'career', 'havilah', 'finances', 'relationships', 'personalGrowth'] as const
   223→  const scored = areas.map(a => ({ area: a, score: latest[a] as number }))
   224→
   225→  const sorted = [...scored].sort((a, b) => a.score - b.score)
   226→  const lowest = sorted.filter(s => s.score <= 3).map(s => s.area)
   227→  const highest = sorted.filter(s => s.score >= 7).map(s => s.area)
   228→
   229→  let trend = 'stable'
   230→  if (ctx.recentScores.length >= 2) {
   231→    const prev = ctx.recentScores[ctx.recentScores.length - 1]
   232→    const diff = latest.overall - prev.overall
   233→    if (diff > 1) trend = 'improving'
   234→    else if (diff < -1) trend = 'declining'
   235→  }
   236→
   237→  const analysis = lowest.length > 0
   238→    ? `Your weakest areas right now: **${lowest.map(a => AREA_LABELS[a]).join(', ')}**. These need urgent attention.`
   239→    : highest.length > 0
   240→    ? `You are strongest in: **${highest.map(a => AREA_LABELS[a]).join(', ')}**. Keep building on this momentum.`
   241→    : 'Your scores are moderate across the board. Time to push for excellence.'
   242→
   243→  return { lowest, highest, trend, analysis }
   244→}

   246→function analyzeFinances(ctx: LifeContext): string {
   247→  if (ctx.weekFinances.count === 0) {
   248→    return 'No financial data this week. Track every naira to stay accountable.'
   249→  }
   250→
   251→  const { received, spent, net } = ctx.weekFinances
   252→  const spendingRatio = received > 0 ? spent / received : 0
   253→
   254→  const parts: string[] = []
   255→  parts.push(`This week: **₦${received.toLocaleString()}** in, **₦${spent.toLocaleString()}** out, **₦${net.toLocaleString()}** net.`)
   256→
   257→  if (net < 0) {
   258→    parts.push('You are spending more than you are earning. This is not sustainable. Identify where the money is going and cut non-essential spending immediately.')
   259→  } else if (spendingRatio > 0.8) {
   260→    parts.push('Your spending-to-income ratio is dangerously high. You should aim to save at least 20% of what comes in.')
   261→  } else if (net > 0) {
   262→    parts.push('You have a positive net flow this week. Good discipline. Now think about where that surplus should go — savings, investment, or giving.')
   263→  }
   264→
   265→  // Top spending categories
   266→  const categoryMap: Record<string, number> = {}
   267→  for (const f of ctx.recentFinances.filter(f => f.type === 'spent')) {
   268→    categoryMap[f.category] = (categoryMap[f.category] || 0) + f.amount
   269→  }
   270→  const topCats = Object.entries(categoryMap).sort((a, b) => b[1] - a[1]).slice(0, 3)
   271→  if (topCats.length > 0) {
   272→    parts.push(`Top spending: ${topCats.map(([k, v]) => `**${k}** (₦${v.toLocaleString()})`).join(', ')}`)
   273→  }
   274→
   275→  return parts.join('\n\n')
   276→}

   278→function analyzeMood(ctx: LifeContext): string {
   279→  if (ctx.recentMoodLogs.length === 0) {
   280→    return 'No mood data recorded yet. Start logging your mood, energy, and focus to unlock pattern detection.'
   281→  }
   282→
   283→  const avgMood = Math.round(ctx.recentMoodLogs.reduce((s, l) => s + l.mood, 0) / ctx.recentMoodLogs.length * 10) / 10
   284→  const avgEnergy = Math.round(ctx.recentMoodLogs.reduce((s, l) => s + l.energy, 0) / ctx.recentMoodLogs.length * 10) / 10
   285→  const avgFocus = Math.round(ctx.recentMoodLogs.reduce((s, l) => s + l.focus, 0) / ctx.recentMoodLogs.length * 10) / 10
   286→
   287→  const parts: string[] = []
   288→  parts.push(`Your averages: Mood **${avgMood}/10**, Energy **${avgEnergy}/10**, Focus **${avgFocus}/10**.`)
   289→
   290→  if (avgMood < 4) {
   291→    parts.push('Your mood has been consistently low. This is a signal — not something to push through. Ask yourself: What is draining you? What would genuinely lift your spirit? Talk to someone you trust.')
   292→  } else if (avgMood >= 7) {
   293→    parts.push('Your mood is strong. Protect the habits and rhythms that produce this. What is working? Do more of it.')
   294→  }
   295→
   296→  if (avgEnergy < 4) {
   297→    parts.push('Your energy is depleted. Sleep more. Eat better. Move your body. Low energy undermines everything else — this is not optional.')
   298→  }
   299→
   300→  if (avgFocus < 4) {
   301→    parts.push('Your focus has been scattered. Consider: Are you trying to do too many things at once? Pick one priority and give it your full attention for 90 minutes.')
   302→  }
   303→
   304→  return parts.join('\n\n')
   305→}

   307→function generateGoalCoaching(ctx: LifeContext): string {
   308→  if (ctx.activeGoals.length === 0) {
   309→    return 'No active goals right now. Set at least one goal to give your days direction. Without goals, drift is inevitable.'
   310→  }
   311→
   312→  const parts: string[] = []
   313→  parts.push(`You have **${ctx.activeGoals.length}** active goal${ctx.activeGoals.length > 1 ? 's' : ''}:`)
   314→
   315→  for (const g of ctx.activeGoals.slice(0, 5)) {
   316→    const done = g.tasks.filter(t => t.status === 'Completed').length
   317→    const total = g.tasks.length
   318→    const pct = total > 0 ? Math.round(done / total * 100) : 0
   319→    parts.push(`- **[${AREA_LABELS[g.area] || g.area}]** ${g.title} — ${pct}% complete (${done}/${total} tasks)`)
   320→  }
   321→
   322→  // Find stalled goals (0% with tasks)
   323→  const stalled = ctx.activeGoals.filter(g => g.tasks.length > 0 && g.tasks.filter(t => t.status === 'Completed').length === 0)
   324→  if (stalled.length > 0) {
   325→    parts.push(`\n**Warning:** ${stalled.length} goal${stalled.length > 1 ? 's' : ''} with zero progress: ${stalled.map(g => g.title).join(', ')}. This is where drift starts. Pick the most important one and complete one task today.`)
   326→  }
   327→
   328→  return parts.join('\n')
   329→}

   342→ * Generate a rich, contextual coaching response from data analysis.
   343→ * This is the heart of the intelligent coaching engine.

   345→function generateCoachingResponse(message: string, ctx: LifeContext, checkInType?: string): string {
   346→  const lowerMessage = message.toLowerCase()
   347→  const scoreAnalysis = analyzeScores(ctx)
   348→
   349→  // === CHECK-IN RESPONSES ===
   350→  if (checkInType) {
   351→    return generateCheckInResponse(checkInType, ctx, scoreAnalysis)
   352→  }
   353→
   354→  // === GREETING ===
   355→  if (lowerMessage.match(/\b(hi|hello|hey|good morning|good evening|good afternoon)\b/)) {
   356→    return generateGreeting(ctx, scoreAnalysis)
   357→  }
   358→
   359→  // === HELP/COACHING ===
   360→  if (lowerMessage.match(/\b(help|guide|coach|advise|suggest|what should|coaching)\b/)) {
   361→    return generateFullCoachingSession(ctx, scoreAnalysis)
   362→  }
   363→
   364→  // === FINANCE ===
   365→  if (lowerMessage.match(/\b(money|finance|spent|received|budget|saving|naira|income|expense|₦)\b/)) {
   366→    return generateFinanceResponse(ctx)
   367→  }
   368→
   369→  // === MOOD/FEELING ===
   370→  if (lowerMessage.match(/\b(mood|feeling|tired|energ|stress|anxi|happy|sad|motivat|drain|overwhelm|exhaust|depress|burn|sleep|rest)\b/)) {
   371→    return generateMoodResponse(ctx, scoreAnalysis)
   372→  }
   373→
   374→  // === GOALS/PROGRESS ===
   375→  if (lowerMessage.match(/\b(goal|progress|track|accomplish|achieve|target|score|drift|move|stuck|stall)\b/)) {
   376→    return generateProgressResponse(ctx, scoreAnalysis)
   377→  }
   378→
   379→  // === FAITH ===
   380→  if (lowerMessage.match(/\b(faith|prayer|devotion|bible|scripture|god|spirit|church|worship)\b/)) {
   381→    return generateAreaSpecificResponse('faith', ctx, scoreAnalysis)
   382→  }
   383→
   384→  // === HEALTH ===
   385→  if (lowerMessage.match(/\b(health|gym|exercise|workout|sleep|diet|food|weight|energy|body)\b/)) {
   386→    return generateAreaSpecificResponse('health', ctx, scoreAnalysis)
   387→  }
   388→
   389→  // === CAREER ===
   390→  if (lowerMessage.match(/\b(career|job|work|cv|resume|interview|skill|professional|application)\b/)) {
   391→    return generateAreaSpecificResponse('career', ctx, scoreAnalysis)
   392→  }
   393→
   394→  // === BUSINESS ===
   395→  if (lowerMessage.match(/\b(havilah|business|entrepreneur|client|revenue|hub|writer)\b/)) {
   396→    return generateAreaSpecificResponse('havilah', ctx, scoreAnalysis)
   397→  }
   398→
   399→  // === RELATIONSHIPS ===
   400→  if (lowerMessage.match(/\b(relationship|family|friend|mentor|community|people|connect)\b/)) {
   401→    return generateAreaSpecificResponse('relationships', ctx, scoreAnalysis)
   402→  }
   403→
   404→  // === DEFAULT: Full coaching overview ===
   405→  return generateDefaultResponse(ctx, scoreAnalysis)
   406→}

   408→function generateCheckInResponse(type: string, ctx: LifeContext, scoreAnalysis: ReturnType<typeof analyzeScores>): string {
   409→  const typeLabels: Record<string, string> = {
   410→    morning: 'Morning Alignment',
   411→    midday: 'Midday Correction',

   854→// GET /api/chat — Retrieve chat history
   855→export async function GET(request: NextRequest) {
   856→  try {
   857→    const { searchParams } = new URL(request.url)
   858→
   859→    // ── Health check sub-path ──
   860→    // GET /api/chat?health=1 — lightweight check if the endpoint is reachable
   861→    // without loading any messages. Used by the chat client to test connectivity.
   862→    if (searchParams.get('health') === '1') {
   863→      return NextResponse.json({ ok: true, timestamp: new Date().toISOString() })
   864→    }
   865→
   866→    const limit = Math.min(parseInt(searchParams.get('limit') || '50', 10), 200)
   867→    const before = searchParams.get('before') // cursor: message ID to paginate before
   868→    // When `latest=true`, return the NEWEST `limit` messages (in asc order for
   869→    // display). Used by the chat client to fetch a server-side fallback after
   870→    // an empty stream — without this flag the endpoint returns the OLDEST
   871→    // `limit` messages, which would miss freshly-saved fallback responses on
   872→    // established accounts.
   873→    const latest = searchParams.get('latest') === 'true'
   874→
   875→    const where = before ? { createdAt: { lt: (await db.chatMessage.findUnique({ where: { id: before } }))?.createdAt ?? new Date() } } : {}
   876→
   877→    let messages
   878→    if (latest) {
   879→      messages = await db.chatMessage.findMany({
   880→        where,
   881→        orderBy: { createdAt: 'desc' },
   882→        take: limit,
   883→      })
   884→      // Reverse to asc so callers can render oldest→newest consistently.
   885→      messages.reverse()
   886→    } else {
   887→      messages = await db.chatMessage.findMany({
   888→        where,
   889→        orderBy: { createdAt: 'asc' },
   890→        take: limit,
   891→      })
   892→    }
   893→
   894→    return NextResponse.json({
   895→      messages: messages.map(m => ({
   896→        id: m.id,
   897→        role: m.role,
   898→        content: m.content,
   899→        checkInType: m.checkInType,
   900→        timestamp: m.createdAt.toISOString(),
   901→      })),
   902→    })
   903→  } catch (error) {
   904→    console.error('Chat GET error:', error)
   905→    return NextResponse.json({ error: 'Failed to fetch chat history' }, { status: 500 })
   906→  }
   907→}

   909→export async function POST(request: NextRequest) {
   910→  try {
   911→    // Rate limiting: 30 messages per minute per IP
   912→    const ip = getClientIp(request)
   913→    const rl = rateLimit(`chat:${ip}`, 30, 60_000)
   914→    if (!rl.allowed) {
   915→      return NextResponse.json(
   916→        { error: 'Too many requests. Please slow down.' },
   917→        { status: 429, headers: { 'Retry-After': String(Math.ceil(rl.resetIn / 1000)) } }
   918→      )
   919→    }
   920→
   921→    const rawBody = await request.json()
   922→    const parsed = ChatMessageSchema.safeParse(rawBody)
   923→    if (!parsed.success) {
   924→      return NextResponse.json({ error: parsed.error.errors[0].message }, { status: 400 })
   925→    }
   926→    const body = parsed.data as ChatRequestBody
   927→    const { checkInType, history, stream: streamMode } = body
   928→    const message = sanitizeForAI(body.message)
   929→    const today = getTodayInTimezone()
   930→
   931→    // Server-side strict mode enforcement for check-ins
   932→    if (checkInType) {
   933→      try {
   934→        const settingsRecord = await db.settings.findUnique({ where: { key: 'user_settings' } })
   935→        if (settingsRecord) {
   936→          const settings = JSON.parse(settingsRecord.value)
   937→          const checkInWindows = settings.checkInWindows || {}
   938→          const { allowed, message: strictMessage } = isWithinCheckInWindowServer(checkInType, {
   939→            strictMode: checkInWindows.strictMode ?? false,
   940→            morningEnabled: checkInWindows.morningEnabled ?? true,
   941→            morningTime: checkInWindows.morningTime ?? '05:00',
   942→            eveningEnabled: checkInWindows.eveningEnabled ?? true,
   943→            eveningTime: checkInWindows.eveningTime ?? '20:30',
   944→            windowMinutes: checkInWindows.windowMinutes ?? 60,
   945→          })
   946→          if (!allowed) {
   947→            return NextResponse.json(
   948→              { error: 'Check-in window closed', message: strictMessage || 'Strict mode is enabled. Please check in during your scheduled window.' },
   949→              { status: 403 }
   950→            )
   951→          }
   952→        }
   953→      } catch (settingsError) {
   954→        console.error('Failed to check strict mode settings:', settingsError)
   955→      }
   956→    }
   957→
   958→    // Store user message — best-effort. A transient DB write failure (e.g.
   959→    // readonly SQLite in some sandboxes, or a momentary PostgreSQL connection
   960→    // hiccup on Vercel) must NOT block the AI response. We log and continue;
   961→    // the user's message simply won't appear in history, which is far better
   962→    // than surfacing a hard 500 "Failed to generate response" to the user.
   963→    let userMessage: { id: string } | null = null
   964→    try {
   965→      userMessage = await db.chatMessage.create({
   966→        data: {
   967→          role: 'user',
   968→          content: message,
   969→          checkInType: checkInType || null,
   970→        },
   971→      })
   972→    } catch (userMsgErr) {
   973→      console.error('Failed to persist user message (continuing anyway):', userMsgErr)
   974→    }
   975→
   976→    // Build context from database (used by both AI and fallback)
   977→    const ctx = await buildContext()
   978→    let assistantContent = ''
   979→    let usedAI = false
   980→
   981→    // Try to get AI response with retry
   982→    try {
   983→      const contextStr = formatContextForAI(ctx)
   984→
   985→      const messages: Array<{ role: 'system' | 'user' | 'assistant'; content: string }> = [
   986→        { role: 'system', content: buildSystemPrompt(
   987→          (await db.settings.findUnique({ where: { key: 'user_name' } }))?.value ? JSON.parse((await db.settings.findUnique({ where: { key: 'user_name' } }))!.value) : '',
   988→          (await db.settings.findUnique({ where: { key: 'os_name' } }))?.value ? JSON.parse((await db.settings.findUnique({ where: { key: 'os_name' } }))!.value) : 'MyOS'
   989→        ) },
   990→      ]
   991→
   992→      if (contextStr) {
   993→        // DeepSeek-V3 (deepseek-chat) has a 64K context window — allow plenty of
   994→        // context so the AI can see all check-ins, entries, and history for the day.
   995→        const truncated = contextStr.length > 24000
   996→          ? contextStr.slice(0, 24000) + '\n[...context truncated for brevity...]'
   997→          : contextStr
   998→        messages.push({
   999→          role: 'system',
  1000→          content: `CURRENT CONTEXT FOR THIS CONVERSATION:\n${truncated}`,
  1001→        })
  1002→      }
  1003→
  1004→      if (history && Array.isArray(history)) {
  1005→        for (const msg of history.slice(-30)) {
  1006→          messages.push({
  1007→            role: msg.role as 'user' | 'assistant',
  1008→            content: msg.content,
  1009→          })
  1010→        }
  1011→      }
  1012→
  1013→      messages.push({ role: 'user', content: message })
  1014→
  1015→      // ── Streaming path ──
  1016→      if (streamMode) {
  1017→        try {
  1018→          const openai = getOpenAIClient()
  1019→
  1020→          // deepseek-chat = DeepSeek-V3. Fast (2–8s), conversational, streams
  1021→          // the first token immediately, and supports temperature for varied
  1022→          // coaching responses. This is the correct model for real-time chat.
  1023→          // (deepseek-reasoner / R1 was used previously — it thinks for 30–120s
  1024→          // before replying, which caused client timeouts and broke memory
  1025→          // persistence because the fire-and-forget smart-sync never ran.)
  1026→          const stream = await openai.chat.completions.create({
  1027→            model: 'deepseek-chat',
  1028→            messages: messages as OpenAI.ChatCompletionMessageParam[],
  1029→            max_tokens: 4000,
  1030→            temperature: 0.7,
  1031→            stream: true,
  1032→          })
  1033→
  1034→          const encoder = new TextEncoder()
  1035→          let fullText = ''
  1036→
  1037→          // We use a promise to track when the stream finishes so we can save
  1038→          // the assistant message to the DB AFTER the response is sent. This is
  1039→          // critical: if we save inside ReadableStream.start(), the DB write may
  1040→          // never complete if the client disconnects mid-stream (mobile, network
  1041→          // drops, tab close). By moving the save to a fire-and-forget promise
  1042→          // that's independent of the stream lifecycle, we ensure the assistant
  1043→          // response is ALWAYS persisted even if the client disconnects.
  1044→          let streamResolve: (text: string) => void
  1045→          const streamDone = new Promise<string>(resolve => { streamResolve = resolve })
  1046→
  1047→          const readable = new ReadableStream({
  1048→            async start(controller) {
  1049→              try {
  1050→                for await (const chunk of stream) {
  1051→                  // Skip internal reasoning tokens — only stream the final answer
  1052→                  const delta = chunk.choices[0]?.delta as { content?: string; reasoning_content?: string }
  1053→                  const content = delta?.content
  1054→                  if (content) {
  1055→                    fullText += content
  1056→                    controller.enqueue(encoder.encode(content))
  1057→                  }
  1058→                }
  1059→                // Signal to the client that the stream completed normally
  1060→                controller.enqueue(encoder.encode('\x00[DONE]'))
  1061→              } catch (streamErr) {
  1062→                console.error('Stream iteration error:', streamErr)
  1063→                // Signal stream error to client
  1064→                controller.enqueue(encoder.encode('\x00[ERROR]'))
  1065→              }
  1066→
  1067→              // Fallback: if the OpenAI stream produced nothing (e.g. it
  1068→              // errored immediately or returned zero chunks), synthesize a
  1069→              // real coaching response and stream it to the client so the
  1070→              // user never sees an empty AI bubble. The fallback is also
  1071→              // persisted to the DB below — mirroring the normal success path.
  1072→              if (!fullText) {
  1073→                let fallback = ''
  1074→                try {
  1075→                  fallback = generateCoachingResponse(message, ctx, checkInType)
  1076→                } catch (genErr) {
  1077→                  console.error('Fallback generateCoachingResponse failed:', genErr)
  1078→                }
  1079→                if (!fallback) {
  1080→                  fallback = "I'm here. Tell me more about what's going on."
  1081→                }
  1082→                fullText = fallback
  1083→                try {
  1084→                  controller.enqueue(encoder.encode(fallback))
  1085→                } catch (enqueueErr) {
  1086→                  console.error('Fallback enqueue error:', enqueueErr)
  1087→                }
  1088→              }
  1089→
  1090→              controller.close()
  1091→              // Resolve the promise with the full text so the DB save
  1092→              // can happen independently of the stream lifecycle.
  1093→              streamResolve(fullText)
  1094→            },
  1095→          })
  1096→
  1097→          // Fire-and-forget: save the assistant message to the DB after the
  1098→          // stream completes. This runs independently of the client connection
  1099→          // state — even if the client disconnects, this promise resolves and
  1100→          // the message is persisted.
  1101→          streamDone.then(async (text) => {
  1102→            if (!text) return
  1103→            try {
  1104→              await db.chatMessage.create({
  1105→                data: {
  1106→                  role: 'assistant',
  1107→                  content: text,
  1108→                  checkInType: checkInType || null,
  1109→                },
  1110→              })
  1111→
  1112→              if (checkInType) {
  1113→                await db.checkIn.upsert({
  1114→                  where: { id: `${today}-${checkInType}` },
  1115→                  create: {
  1116→                    id: `${today}-${checkInType}`,
  1117→                    type: checkInType,
  1118→                    date: today,
  1119→                    data: JSON.stringify({ message }),
  1120→                    aiResponse: JSON.stringify({ response: text }),
  1121→                  },
  1122→                  update: {
  1123→                    data: JSON.stringify({ message }),
  1124→                    aiResponse: JSON.stringify({ response: text }),
  1125→                  },
  1126→                })
  1127→              }
  1128→
  1129→              // Fire-and-forget smart-sync
  1130→              fetch(new URL('/api/smart-sync', request.url), {
  1131→                method: 'POST',
  1132→                headers: { 'Content-Type': 'application/json' },
  1133→                body: JSON.stringify({ message, checkInType }),
  1134→              }).catch(syncErr => console.error('Smart-sync error in streaming chat:', syncErr))
  1135→
  1136→              // Fire-and-forget streak recalculation
  1137→              recalcStreaks(checkInType || undefined).catch(() => {})
  1138→            } catch (dbErr) {
  1139→              console.error('Post-stream DB save error:', dbErr)
  1140→            }
  1141→          }).catch(err => {
  1142→            console.error('Stream done promise error:', err)
  1143→          })
  1144→
  1145→          return new Response(readable, {
  1146→            headers: {
  1147→              'Content-Type': 'text/plain; charset=utf-8',
  1148→              'Cache-Control': 'no-cache, no-transform',
  1149→              'X-Accel-Buffering': 'no',
  1150→              'Transfer-Encoding': 'chunked',
  1151→            },
  1152→          })
  1153→        } catch (streamInitError) {
  1154→          console.error('Streaming init failed, falling back to non-streaming:', streamInitError)
  1155→          // Fall through to non-streaming path below
  1156→        }
  1157→      }
  1158→
  1159→      // ── Non-streaming path ──
  1160→      // deepseek-chat (V3) supports temperature — use 0.7 for varied, natural
  1161→      // coaching responses instead of robotic deterministic output.
  1162→      const aiResponse = await callZAIWithRetry(messages, 2, { temperature: 0.7 })
  1163→      if (aiResponse) {
  1164→        assistantContent = aiResponse
  1165→        usedAI = true
  1166→      }
  1167→    } catch (aiError) {
  1168→      console.error('AI generation failed, using intelligent fallback:', aiError)
  1169→    }
  1170→
  1171→    // Use intelligent coaching engine if AI didn't produce a response
  1172→    if (!usedAI) {
  1173→      assistantContent = generateCoachingResponse(message, ctx, checkInType)
  1174→    }
  1175→
  1176→    // Store assistant message — best-effort (mirror of user-message handling).
  1177→    // If the DB write fails we still return the generated response to the
  1178→    // client rather than 500ing.
  1179→    let assistantMessage: { id: string } | null = null
  1180→    try {
  1181→      assistantMessage = await db.chatMessage.create({
  1182→        data: {
  1183→          role: 'assistant',
  1184→          content: assistantContent,
  1185→          checkInType: checkInType || null,
  1186→        },
  1187→      })
  1188→    } catch (assistantMsgErr) {
  1189→      console.error('Failed to persist assistant message (returning response anyway):', assistantMsgErr)
  1190→    }
  1191→
  1192→    // Store check-in if type is provided — best-effort. A DB failure here
  1193→    // must not prevent the response from being returned.
  1194→    if (checkInType) {
  1195→      try {
  1196→        await db.checkIn.upsert({
  1197→          where: { id: `${today}-${checkInType}` },
  1198→          create: {
  1199→            id: `${today}-${checkInType}`,
  1200→            type: checkInType,
  1201→            date: today,
  1202→            data: JSON.stringify({ message }),
  1203→            aiResponse: JSON.stringify({ response: assistantContent }),
  1204→          },
  1205→          update: {
  1206→            data: JSON.stringify({ message }),
  1207→            aiResponse: JSON.stringify({ response: assistantContent }),
  1208→          },
  1209→        })
  1210→      } catch (checkInErr) {
  1211→        console.error('Failed to persist check-in (returning response anyway):', checkInErr)
  1212→      }
  1213→      // Recalculate streaks after check-in
  1214→      recalcStreaks(checkInType).catch(() => {})
  1215→    }
  1216→
  1217→    // Smart-sync
  1218→    let syncResult = { goalsUpdated: 0, tasksUpdated: 0, financeEntriesCreated: 0 }
  1219→    try {
  1220→      const syncRes = await fetch(new URL('/api/smart-sync', request.url), {
  1221→        method: 'POST',
  1222→        headers: { 'Content-Type': 'application/json' },
  1223→        body: JSON.stringify({ message, checkInType }),
  1224→      })
  1225→      if (syncRes.ok) {
  1226→        const syncData = await syncRes.json()
  1227→        if (syncData.applied) {
  1228→          syncResult = syncData.applied
  1229→        }
  1230→      }
  1231→    } catch (syncError) {
  1232→      console.error('Smart-sync error in chat:', syncError)
  1233→    }
  1234→
  1235→    return NextResponse.json({
  1236→      response: assistantContent,
  1237→      messageId: assistantMessage?.id ?? null,
  1238→      userMessageId: userMessage?.id ?? null,
  1239→      syncResult,
  1240→      _meta: { usedAI },
  1241→    })
  1242→  } catch (error) {
  1243→    console.error('Chat API error:', error)
  1244→    return NextResponse.json(
  1245→      { error: 'Failed to generate response' },
  1246→      { status: 500 }
  1247→    )
  1248→  }
  1249→}
