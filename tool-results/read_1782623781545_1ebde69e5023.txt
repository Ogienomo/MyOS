    56→// Frontend deduplication: filter out goals with identical titles within the same area
    57→function deduplicateGoals(goals: Goal[]): { goals: Goal[]; duplicates: Goal[] } {
    58→  const seen = new Map<string, Goal>()
    59→  const deduped: Goal[] = []
    60→  const duplicates: Goal[] = []
    61→
    62→  for (const goal of goals) {
    63→    const key = `${goal.area}:${goal.title.toLowerCase().trim()}`
    64→    if (seen.has(key)) {
    65→      duplicates.push(goal)
    66→    } else {
    67→      seen.set(key, goal)
    68→      deduped.push(goal)
    69→    }
    70→  }
    71→
    72→  return { goals: deduped, duplicates }
    73→}
